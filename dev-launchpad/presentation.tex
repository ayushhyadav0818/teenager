% Slide 2: The Problem
\begin{frame}{The Problem Statement}
    \begin{itemize}
        \item \textbf{High Entry Barrier:} Developers build side-projects daily, but struggle to find early adopters.
        \item \textbf{Scattered Feedback:} Feedbacks are scattered across Twitter, Reddit, and Discord.
        \item \textbf{Missing Validation:} Lack of a centralized platform specifically for micro-SaaS and dev-oriented tools.
    \end{itemize}
\end{frame}

% Slide 3: The Solution (DevLaunch)
\begin{frame}{The Solution: DevLaunch}
    \begin{block}{Our Vision}
        A premium curated showcase platform built for developers, by developers.
    \end{block}
    \begin{itemize}
        \item \textbf{Upvoting System:} Crowd-sourced rating mechanism for validating startup potential.
        \item \textbf{Direct Linkage:} Github source URLs and live link integrations out-of-the-box.
        \item \textbf{Interactive Discussions:} Live forum/comments section underneath every startup product.
    \end{itemize}
\end{frame}
